## 功能

- 箭头导航应选择块内的标题。

</PackageInfo>

## 安装

```bash
npm install @udecode/plate-caption
```

## 使用

```tsx
import { CaptionPlugin } from '@udecode/plate-caption/react';
import {
  AudioPlugin,
  FilePlugin,
  ImagePlugin,
  MediaEmbedPlugin,
  VideoPlugin,
} from '@udecode/plate-media/react';
```

```tsx
const plugins = [
  // ...otherPlugins,
  ImagePlugin,
  VideoPlugin,
  AudioPlugin,
  FilePlugin,
  MediaEmbedPlugin,
  CaptionPlugin.configure({
    options: {
      plugins: [
        ImagePlugin,
        VideoPlugin,
        AudioPlugin,
        FilePlugin,
        MediaEmbedPlugin,
      ],
    },
  }),
];
```

## 插件

### CaptionPlugin

<APIOptions>
  <APIItem name="plugins" type="string[]" required>
    具有标题的块的插件键。
  </APIItem>
  <APIItem name="focusEndPath" type="Path" optional>
    聚焦到标题末尾的路径。
  </APIItem>
  <APIItem name="focusStartPath" type="Path" optional>
    聚焦到标题开头的路径。
  </APIItem>
  <APIItem name="visibleId" type="string" optional>
    可见标题的 ID。
  </APIItem>
  <APIItem name="isVisible" type="boolean" optional>
    标题是否可见。
  </APIItem>
</APIOptions>
