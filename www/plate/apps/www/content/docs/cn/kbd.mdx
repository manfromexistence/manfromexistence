## 安装

```bash
npm install @udecode/plate-kbd
```

## 使用

```tsx
import { KbdPlugin } from '@udecode/plate-kbd';
```

```tsx
const plugins = [
  // ...otherPlugins,
  KbdPlugin,
];
```

```tsx
const components ={
  [KbdPlugin.key]: KbdLeaf,
}
```
